⚠️ NULL — à comprendre
⚠️
NULL ne signifie pas 0 ou vide — c'est l'absence totale de valeur. NULL ≠ NULL, donc toujours utiliser IS NULL et non = NULL.
-- ✗ Ne fonctionne PAS
WHERE email = NULL

-- ✓ Correct
WHERE email IS NULL
WHERE email IS NOT NULL

-- NULL dans les calculs = NULL
SELECT 5 + NULL; -- → NULL

DDL — Langage de définition de données

CREATE · ALTER · DROP · TRUNCATE — pour gérer la structure
CREATE TABLECréer
syntaxe complète
CREATE TABLE utilisateurs (
  id         INT           PRIMARY KEY AUTO_INCREMENT,
  nom        VARCHAR(100) NOT NULL,
  email      VARCHAR(255) UNIQUE NOT NULL,
  age        INT           CHECK (age >= 18),
  role       ENUM('admin','user') DEFAULT 'user',
  created_at TIMESTAMP     DEFAULT CURRENT_TIMESTAMP
);
ALTER TABLEModifier
opérations possibles
-- Ajouter une colonne
ALTER TABLE utilisateurs
  ADD COLUMN telephone VARCHAR(20);

-- Modifier une colonne
ALTER TABLE utilisateurs
  MODIFY COLUMN nom VARCHAR(200);

-- Renommer une colonne (MySQL 8+)
ALTER TABLE utilisateurs
  RENAME COLUMN nom TO nom_complet;

-- Supprimer une colonne
ALTER TABLE utilisateurs
  DROP COLUMN telephone;

-- Ajouter une clé étrangère
ALTER TABLE commandes
  ADD CONSTRAINT fk_user
  FOREIGN KEY (user_id)
  REFERENCES utilisateurs(id);
DROP & TRUNCATESupprimer
-- Supprimer une table entière
DROP TABLE utilisateurs;

-- Supprimer si elle existe (pas d'erreur)
DROP TABLE IF EXISTS utilisateurs;

-- Vider la table (garde la structure)
-- Plus rapide que DELETE sans WHERE
TRUNCATE TABLE utilisateurs;

-- Supprimer la base entière
DROP DATABASE ma_base;
🚨
DROP et TRUNCATE sont irréversibles sans sauvegarde. Toujours faire un backup avant !

DML — Manipulation des données

INSERT · UPDATE · DELETE — pour créer, modifier, supprimer des données
INSERT — Insérer
-- Insertion simple
INSERT INTO utilisateurs (nom, email)
VALUES ('Alice', 'alice@exemple.com');

-- Insertion multiple (plus performant)
INSERT INTO utilisateurs (nom, email)
VALUES
  ('Bob', 'bob@exemple.com'),
  ('Clara', 'clara@exemple.com'),
  ('David', 'david@exemple.com');

-- Insérer depuis une autre table
INSERT INTO archives (nom, email)
SELECT nom, email FROM utilisateurs
WHERE created_at < '2023-01-01';
UPDATE — Modifier
-- Modifier un enregistrement
UPDATE utilisateurs
SET email = 'newemail@exemple.com'
WHERE id = 42;

-- Modifier plusieurs colonnes
UPDATE produits
SET
  prix = prix * 1.10,   -- +10%
  stock = stock - 5
WHERE categorie = 'electronique';

-- UPDATE avec JOIN
UPDATE commandes c
JOIN utilisateurs u ON c.user_id = u.id
SET c.statut = 'verifie'
WHERE u.role = 'premium';
🚨
Sans clause WHERE, UPDATE modifie toutes les lignes !
DELETE — Supprimer
-- Supprimer un enregistrement
DELETE FROM utilisateurs
WHERE id = 42;

-- Supprimer selon une condition
DELETE FROM sessions
WHERE expires_at < NOW();

-- Supprimer avec sous-requête
DELETE FROM commandes
WHERE user_id IN (
  SELECT id FROM utilisateurs
  WHERE actif = 0
);

-- ⚠️ Supprimer TOUT (éviter, préférer TRUNCATE)
DELETE FROM logs;
UPSERT — Insérer ou Mettre à jour
-- MySQL : INSERT ... ON DUPLICATE KEY
INSERT INTO preferences (user_id, theme)
VALUES (1, 'dark')
ON DUPLICATE KEY UPDATE
  theme = VALUES(theme);

-- PostgreSQL : INSERT ... ON CONFLICT
INSERT INTO preferences (user_id, theme)
VALUES (1, 'dark')
ON CONFLICT (user_id)
DO UPDATE SET theme = EXCLUDED.theme;

-- Ne rien faire si conflit
INSERT INTO tags (nom) VALUES ('sql')
ON CONFLICT DO NOTHING;

SELECT & Filtres

La commande la plus utilisée — lire et filtrer les données
SELECT — Structure complète
SELECT
  u.id,
  u.nom,
  u.email,
  COUNT(c.id)     AS nb_commandes,
  SUM(c.montant) AS total
FROM utilisateurs u
LEFT JOIN commandes c
  ON u.id = c.user_id
WHERE u.actif = 1
GROUP BY u.id
HAVING SUM(c.montant) > 1000
ORDER BY total DESC
LIMIT 10;
WHERE — Filtres
-- Comparaisons
WHERE age >= 18
WHERE statut != 'inactif'   -- ou <>

-- Plage de valeurs
WHERE prix BETWEEN 10 AND 100
WHERE date NOT BETWEEN '2024-01-01' AND '2024-12-31'

-- Liste de valeurs
WHERE pays IN ('FR', 'BE', 'CH')
WHERE id NOT IN (1, 2, 3)

-- Texte (LIKE)
WHERE nom LIKE 'Al%'       -- commence par Al
WHERE email LIKE '%@gmail%'-- contient @gmail
WHERE code LIKE 'A_B'      -- _ = 1 caractère

-- Combinaisons logiques
WHERE age > 18 AND pays = 'FR'
WHERE role = 'admin' OR role = 'superadmin'
WHERE NOT actif = 0
ORDER BY · LIMIT · OFFSET
-- Trier par une colonne
ORDER BY nom ASC      -- croissant (défaut)
ORDER BY prix DESC    -- décroissant

-- Trier par plusieurs colonnes
ORDER BY pays ASC, nom ASC

-- Limiter le nombre de résultats
LIMIT 10

-- Pagination : page 3 avec 10 items/page
LIMIT 10 OFFSET 20   -- saute les 20 premiers

-- NULL en dernier lors du tri
ORDER BY date DESC NULLS LAST  -- PostgreSQL

-- DISTINCT : supprimer les doublons
SELECT DISTINCT pays FROM utilisateurs;
CASE — Condition dans SELECT
-- CASE simple
SELECT nom,
  CASE role
    WHEN 'admin'     THEN 'Administrateur'
    WHEN 'user'      THEN 'Utilisateur'
    ELSE 'Inconnu'
  END AS role_label
FROM utilisateurs;

-- CASE avec conditions
SELECT produit,
  CASE
    WHEN stock = 0    THEN 'Rupture'
    WHEN stock < 10   THEN 'Stock faible'
    ELSE 'Disponible'
  END AS disponibilite
FROM produits;

-- CASE dans ORDER BY
ORDER BY
  CASE priorite
    WHEN 'haute' THEN 1
    WHEN 'moyenne' THEN 2
    ELSE 3
  END;

JOIN — Jointures

Combiner les données de plusieurs tables
📐
Exemple : tables utilisateurs (u) et commandes (c) liées par u.id = c.user_id
INNER JOINIntersection
💡
Retourne uniquement les lignes ayant une correspondance dans les DEUX tables.
SELECT u.nom, c.montant
FROM utilisateurs u
INNER JOIN commandes c
  ON u.id = c.user_id;
-- Seuls les utilisateurs AVEC commandes
LEFT JOINTout à gauche
💡
Retourne TOUTES les lignes de gauche + les correspondances de droite (NULL si aucune).
SELECT u.nom, c.montant
FROM utilisateurs u
LEFT JOIN commandes c
  ON u.id = c.user_id;
-- Tous les utilisateurs, même sans commande
-- (montant = NULL si aucune commande)

-- Astuce : trouver les utilisateurs SANS commande
WHERE c.id IS NULL
RIGHT JOINTout à droite
SELECT u.nom, c.montant
FROM utilisateurs u
RIGHT JOIN commandes c
  ON u.id = c.user_id;
-- Toutes les commandes, même orphelines
-- (cas rare, préférer LEFT JOIN inversé)
FULL OUTER JOINTout
-- PostgreSQL / SQL Server
SELECT u.nom, c.montant
FROM utilisateurs u
FULL OUTER JOIN commandes c
  ON u.id = c.user_id;
-- Tout de gauche + tout de droite

-- MySQL : émulation avec UNION
SELECT * FROM u LEFT JOIN c ON u.id=c.uid
UNION
SELECT * FROM u RIGHT JOIN c ON u.id=c.uid;
SELF JOIN & CROSS JOIN
-- SELF JOIN : une table jointe sur elle-même
-- Ex : trouver les employés et leur manager
SELECT
  e.nom AS employe,
  m.nom AS manager
FROM employes e
LEFT JOIN employes m
  ON e.manager_id = m.id;

-- CROSS JOIN : produit cartésien (toutes combinaisons)
SELECT t.taille, c.couleur
FROM tailles t
CROSS JOIN couleurs c;
-- S·M·L × Rouge·Bleu·Vert = 9 lignes
JOIN multiple & conditions
-- Joindre 3+ tables
SELECT
  u.nom, c.date, p.nom AS produit
FROM utilisateurs u
JOIN commandes c  ON u.id = c.user_id
JOIN lignes l      ON c.id = l.commande_id
JOIN produits p   ON l.produit_id = p.id
WHERE c.statut = 'livree';

-- JOIN avec condition multiple
JOIN tarifs t
  ON p.id = t.produit_id
  AND t.actif = 1

GROUP BY & Fonctions d'agrégation

Regrouper et calculer des statistiques sur des ensembles de lignes
Fonctions d'agrégation
FonctionRésultatNote
COUNT(*)Nombre de lignes totalInclut les NULL
COUNT(col)Lignes non-NULLExclut les NULL
COUNT(DISTINCT col)Valeurs uniques–
SUM(col)SommeIgnore les NULL
AVG(col)MoyenneIgnore les NULL
MIN(col)Valeur minimale–
MAX(col)Valeur maximale–
GROUP_CONCAT(col)Concatène en chaîneMySQL
STRING_AGG(col,',')Concatène en chaînePostgreSQL
GROUP BY — Exemples
-- Ventes par catégorie
SELECT
  categorie,
  COUNT(*)       AS nb_produits,
  AVG(prix)       AS prix_moyen,
  MAX(prix)       AS prix_max,
  SUM(stock)      AS stock_total
FROM produits
GROUP BY categorie
ORDER BY stock_total DESC;

-- Grouper par plusieurs colonnes
SELECT annee, mois, SUM(montant)
FROM ventes
GROUP BY annee, mois;
HAVING — Filtrer les groupes
📌
WHERE filtre les lignes AVANT le groupement.
HAVING filtre les groupes APRÈS l'agrégation.
-- Catégories avec plus de 10 produits
SELECT categorie, COUNT(*) AS nb
FROM produits
GROUP BY categorie
HAVING COUNT(*) > 10;

-- WHERE + HAVING combinés
SELECT user_id, SUM(montant) AS total
FROM commandes
WHERE statut = 'validee'       -- filtre lignes
GROUP BY user_id
HAVING SUM(montant) > 500;   -- filtre groupes

Sous-requêtes (Subqueries)

Imbriquer des requêtes pour des recherches complexes
Sous-requête dans WHERE
-- Utilisateurs ayant passé au moins une commande
SELECT nom FROM utilisateurs
WHERE id IN (
  SELECT DISTINCT user_id FROM commandes
);

-- Produits plus chers que la moyenne
SELECT nom, prix FROM produits
WHERE prix > (
  SELECT AVG(prix) FROM produits
);

-- EXISTS : vérifie l'existence
SELECT nom FROM utilisateurs u
WHERE EXISTS (
  SELECT 1 FROM commandes c
  WHERE c.user_id = u.id
);
Sous-requête dans FROM
-- Sous-requête comme table temporaire
SELECT stats.categorie, stats.nb
FROM (
  SELECT
    categorie,
    COUNT(*) AS nb,
    AVG(prix) AS prix_moy
  FROM produits
  GROUP BY categorie
) AS stats
WHERE stats.prix_moy > 50;

-- Scalar subquery dans SELECT
SELECT
  nom,
  (SELECT COUNT(*) FROM commandes
   WHERE user_id = u.id) AS nb_cmd
FROM utilisateurs u;
ANY · ALL · SOME
-- Prix supérieur à AU MOINS UN prix de table2
WHERE prix > ANY (
  SELECT prix FROM promo
)

-- Prix supérieur à TOUS les prix de table2
WHERE prix > ALL (
  SELECT prix FROM promo
)

-- SOME = ANY (synonyme)
WHERE note = SOME (
  SELECT note_min FROM criteres
)
UNION & INTERSECT
-- UNION : combine sans doublons
SELECT nom FROM clients_fr
UNION
SELECT nom FROM clients_be;

-- UNION ALL : garde les doublons (plus rapide)
SELECT email FROM newsletter
UNION ALL
SELECT email FROM inscrits;

-- INTERSECT : valeurs dans les DEUX (PostgreSQL)
SELECT id FROM table1
INTERSECT
SELECT id FROM table2;

-- EXCEPT : dans table1 mais PAS dans table2
SELECT id FROM table1
EXCEPT
SELECT id FROM table2;

CTE — Common Table Expressions (WITH)

Des requêtes nommées et réutilisables, bien plus lisibles que les sous-requêtes
CTE simple
-- Syntaxe de base
WITH clients_premium AS (
  SELECT id, nom, total_achats
  FROM utilisateurs
  WHERE total_achats > 5000
)
SELECT *
FROM clients_premium
ORDER BY total_achats DESC;

-- CTEs multiples
WITH
  ventes_2024 AS (
    SELECT * FROM ventes
    WHERE YEAR(date) = 2024
  ),
  top_produits AS (
    SELECT produit_id, SUM(qte) total
    FROM ventes_2024   -- réutilise la 1re CTE
    GROUP BY produit_id
  )
SELECT p.nom, tp.total
FROM top_produits tp
JOIN produits p ON p.id = tp.produit_id
ORDER BY tp.total DESC;
CTE Récursive
🔁
Indispensable pour naviguer dans des structures hiérarchiques (arborescences, menus, organigrammes).
-- Hiérarchie d'employés (organigramme)
WITH RECURSIVE hierarchie AS (
  -- Ancre : point de départ (PDG)
  SELECT id, nom, manager_id, 0 AS niveau
  FROM employes
  WHERE manager_id IS NULL

  UNION ALL

  -- Récursion : sous-employés
  SELECT e.id, e.nom, e.manager_id, h.niveau+1
  FROM employes e
  JOIN hierarchie h
    ON e.manager_id = h.id
)
SELECT REPEAT('  ', niveau) || nom AS arbre
FROM hierarchie
ORDER BY niveau;

Fonctions de fenêtre (Window Functions)

Calculs sur un ensemble de lignes sans les regrouper — OVER (PARTITION BY)
💡
Contrairement à GROUP BY, les window functions conservent chaque ligne. La "fenêtre" définit quelles lignes sont prises en compte pour le calcul.
OVER — Syntaxe
-- Total des ventes par département (garde chaque ligne)
SELECT
  nom,
  departement,
  salaire,
  SUM(salaire) OVER (
    PARTITION BY departement
  ) AS total_dept,
  AVG(salaire) OVER (
    PARTITION BY departement
  ) AS moy_dept
FROM employes;

-- Total global (sans PARTITION)
SUM(salaire) OVER () AS total_global
ROW_NUMBER · RANK · DENSE_RANK
SELECT
  nom, departement, salaire,
  ROW_NUMBER() OVER (
    PARTITION BY departement
    ORDER BY salaire DESC
  ) AS rang_unique,

  RANK() OVER (
    PARTITION BY departement
    ORDER BY salaire DESC
  ) AS rang_ex_aequo,
  -- Ex-aequos reçoivent le même rang, suivant = rang+nb

  DENSE_RANK() OVER (
    PARTITION BY departement
    ORDER BY salaire DESC
  ) AS rang_dense
  -- Pas de "trous" dans la numérotation
FROM employes;
LAG · LEAD · NTILE
SELECT
  date, ventes,

  -- Valeur de la ligne précédente
  LAG(ventes) OVER (ORDER BY date)
    AS ventes_hier,

  -- Valeur de la ligne suivante
  LEAD(ventes) OVER (ORDER BY date)
    AS ventes_demain,

  -- Variation jour/jour
  ventes - LAG(ventes) OVER (ORDER BY date)
    AS variation,

  -- Diviser en 4 quartiles
  NTILE(4) OVER (ORDER BY ventes)
    AS quartile

FROM ventes_daily;
FIRST_VALUE · LAST_VALUE · RUNNING TOTAL
SELECT
  mois, montant,

  -- Première valeur de la partition
  FIRST_VALUE(montant)
    OVER (ORDER BY mois) AS premier_mois,

  -- Cumul progressif (running total)
  SUM(montant) OVER (
    ORDER BY mois
    ROWS BETWEEN UNBOUNDED PRECEDING
    AND CURRENT ROW
  ) AS cumul,

  -- Moyenne mobile sur 3 mois
  AVG(montant) OVER (
    ORDER BY mois
    ROWS BETWEEN 2 PRECEDING AND CURRENT ROW
  ) AS moy_3mois

FROM revenus_mensuels;

Index & Performance

Accélérer les requêtes — un bon index peut multiplier les performances par 100x
CREATE INDEX
-- Index simple
CREATE INDEX idx_email
  ON utilisateurs(email);

-- Index unique
CREATE UNIQUE INDEX idx_email_uniq
  ON utilisateurs(email);

-- Index composite (ordre important !)
CREATE INDEX idx_user_date
  ON commandes(user_id, created_at);

-- Supprimer un index
DROP INDEX idx_email ON utilisateurs;

-- Voir les index d'une table
SHOW INDEX FROM utilisateurs;  -- MySQL
SELECT * FROM pg_indexes
  WHERE tablename='utilisateurs'; -- PG
EXPLAIN — Analyser une requête
-- Voir le plan d'exécution
EXPLAIN
SELECT * FROM utilisateurs
WHERE email = 'alice@exemple.com';

-- Avec stats (PostgreSQL)
EXPLAIN ANALYZE
SELECT * FROM commandes
WHERE user_id = 42;

-- Ce qu'on cherche dans EXPLAIN :
-- ✓ "Using index" = bon !
-- ✗ "Full table scan" = mauvais, ajouter un index
-- ✗ "Using filesort" = tri coûteux
💡
Indexer les colonnes utilisées dans WHERE, JOIN ON, ORDER BY, et GROUP BY.
📐 Règles d'optimisation
✓ Faire✗ Éviter
Indexer les clés étrangères (FK)SELECT * en production
Utiliser LIMIT sur les grosses tablesFonctions sur colonnes indexées dans WHERE (WHERE YEAR(date) = 2024)
Favoriser EXISTS plutôt que COUNT(*) > 0Sous-requêtes corrélées sur chaque ligne
Utiliser des types appropriés (INT pour IDs)Stocker les nombres en VARCHAR
Colonnes sélectives en premier dans les index compositesTrop d'index (ralentissent les INSERT/UPDATE)
Analyser régulièrement avec EXPLAINOR sur des colonnes indexées (souvent non-optimal)

Transactions (TCL)

Garantir l'intégrité des données — tout ou rien
COMMIT · ROLLBACK · SAVEPOINT
-- Transaction classique (virement bancaire)
START TRANSACTION;  -- ou BEGIN;

-- Débiter le compte émetteur
UPDATE comptes
SET solde = solde - 500
WHERE id = 1;

-- Créditer le destinataire

Vues, Procédures stockées & Triggers

Encapsuler et automatiser la logique SQL
VIEW — Vues
-- Créer une vue
CREATE VIEW v_clients_actifs AS
  SELECT id, nom, email
  FROM utilisateurs
  WHERE actif = 1;

-- Utiliser la vue comme une table
SELECT * FROM v_clients_actifs
WHERE email LIKE '%gmail%';

-- Remplacer une vue existante
CREATE OR REPLACE VIEW v_clients_actifs AS
  SELECT id, nom, email, pays
  FROM utilisateurs
  WHERE actif = 1;

-- Supprimer une vue
DROP VIEW IF EXISTS v_clients_actifs;
Procédure stockée (MySQL)
-- Créer une procédure
DELIMITER //
CREATE PROCEDURE GetUserOrders(IN p_user_id INT)
BEGIN
  SELECT c.*, u.nom
  FROM commandes c
  JOIN utilisateurs u ON c.user_id = u.id
  WHERE c.user_id = p_user_id;
END//
DELIMITER ;

-- Appeler la procédure
CALL GetUserOrders(42);

-- Supprimer
DROP PROCEDURE IF EXISTS GetUserOrders;
TRIGGER — Déclencheur
-- Horodater automatiquement la mise à jour
CREATE TRIGGER trg_update_time
BEFORE UPDATE ON articles
FOR EACH ROW
BEGIN
  SET NEW.updated_at = NOW();
END;

-- Journaliser les suppressions
CREATE TRIGGER trg_log_delete
AFTER DELETE ON commandes
FOR EACH ROW
BEGIN
  INSERT INTO audit_log
    (action, table_name, record_id, date)
  VALUES
    ('DELETE', 'commandes', OLD.id, NOW());
END;
DCL — Permissions
-- Créer un utilisateur
CREATE USER 'alice'@'localhost'
  IDENTIFIED BY 'motdepasse';

-- Accorder des droits
GRANT SELECT, INSERT
  ON ma_base.*
  TO 'alice'@'localhost';

-- Tous les droits
GRANT ALL PRIVILEGES ON ma_base.*
  TO 'bob'@'%';

-- Révoquer des droits
REVOKE INSERT ON ma_base.*
  FROM 'alice'@'localhost';

-- Appliquer les changements
FLUSH PRIVILEGES;

Erreurs courantes & Solutions

Les pièges les plus fréquents pour les débutants
❌ Pièges classiques
-- ✗ ERREUR : UPDATE sans WHERE
UPDATE users SET actif = 0;
-- Désactive TOUS les utilisateurs !
-- ✓ Toujours ajouter WHERE
UPDATE users SET actif=0 WHERE id=5;

-- ✗ ERREUR : = NULL
WHERE email = NULL   -- ne fonctionne pas
-- ✓ Correct
WHERE email IS NULL

-- ✗ ERREUR : colonne dans WHERE mais pas dans GROUP BY
SELECT categorie, nom, COUNT(*)
FROM produits
GROUP BY categorie;  -- nom non agrégé !
-- ✓ Ajouter nom au GROUP BY ou à une agrégation

-- ✗ ERREUR : alias dans WHERE
SELECT prix*1.2 AS prix_ttc
FROM produits
WHERE prix_ttc > 100;  -- ✗ l'alias n'existe pas encore
-- ✓ Répéter l'expression ou utiliser HAVING/sous-requête
⚠️ Autres erreurs fréquentes
ErreurCause & Solution
Duplicate entryViolation de UNIQUE ou PK. Utiliser INSERT IGNORE ou ON DUPLICATE KEY.
Cannot add FK constraintLes types des colonnes liées ne correspondent pas. Vérifier INT vs BIGINT.
Column ambiguousMême nom dans deux tables jointes. Ajouter le préfixe : u.nom

Ordre d'exécution SQL

Comprendre l'ordre dans lequel SQL traite une requête SELECT
L'ordre d'écriture ≠ l'ordre d'exécution
#ClauseCe qui se passeExemple
1FROMIdentifier les tables sourcesFROM utilisateurs u
2JOINJoindre les tablesLEFT JOIN commandes c ON ...
3WHEREFiltrer les lignes (avant agrégation)WHERE u.actif = 1
4GROUP BYRegrouper les lignesGROUP BY u.id
5HAVINGFiltrer les groupes (après agrégation)HAVING COUNT(*) > 2
6SELECTCalculer les colonnes résultantesSELECT nom, COUNT(*)
7DISTINCTSupprimer les doublonsSELECT DISTINCT ...
8ORDER BYTrier les résultats (peut utiliser les alias)ORDER BY total DESC
9LIMIT / OFFSETLimiter le nombre de résultatsLIMIT 10 OFFSET 20
⚠️
C'est pourquoi un alias défini dans SELECT ne peut pas être utilisé dans WHERE (WHERE est exécuté avant SELECT). Par contre, il peut être utilisé dans ORDER BY et HAVING (selon le SGBD).
